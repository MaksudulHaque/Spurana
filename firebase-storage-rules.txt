rules_version = '2';

// Spurana ✦ Firebase Storage rules
//
// Apply via Firebase Console → Storage → Rules tab → paste this → Publish
//
// Schema:
//   /avatars/{uid}                       ← profile pictures (any user can read, only owner writes)
//   /voice/{uid}/{timestamp}.{ext}       ← voice messages (any auth user can read, only owner writes)
//   /media/{uid}/{timestamp}_{filename}  ← attached images/files (same)
//
// Constraints:
//   · Max 10MB per avatar
//   · Max 25MB per voice message
//   · Max 50MB per media file
//   · Only authenticated users can read at all
//   · Only the owner (matching uid in path) can write

service firebase.storage {
  match /b/{bucket}/o {

    // ── Avatars: shared visibility, owner-only writes ──
    match /avatars/{uid} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && request.auth.uid == uid
                   && request.resource.size < 10 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
      allow delete: if request.auth != null && request.auth.uid == uid;
    }

    // ── Voice messages: shared visibility, owner-only writes ──
    match /voice/{uid}/{filename} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && request.auth.uid == uid
                   && request.resource.size < 25 * 1024 * 1024
                   && (request.resource.contentType.matches('audio/.*')
                       || request.resource.contentType.matches('video/.*'));
      allow delete: if request.auth != null && request.auth.uid == uid;
    }

    // ── Media attachments: shared visibility, owner-only writes ──
    match /media/{uid}/{filename} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && request.auth.uid == uid
                   && request.resource.size < 50 * 1024 * 1024;
      allow delete: if request.auth != null && request.auth.uid == uid;
    }

    // Anything else — deny by default
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
